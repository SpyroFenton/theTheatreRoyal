-- Query to list all shows
select * FROM showProduction

-- Query to search by show name
SELECT
showProduction.showName,
showProduction.duration,
performance.showDate,
performance.showStartTime
FROM showProduction
INNER JOIN performance
ON showProduction.id = performance.showProductionID  
WHERE showProduction.showName = ?;

-- Query to search by show date
SELECT
showProduction.showName,
showProduction.duration,
performance.showDate,
performance.showStartTime
FROM showProduction
INNER JOIN performance
ON showProduction.id = performance.showProductionID  
WHERE performance.showDate = ?;
